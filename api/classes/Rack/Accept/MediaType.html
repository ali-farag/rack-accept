25:     def matches(media_type)
26:       type, subtype, params = parse_media_type(media_type)
27:       values.select {|v|
28:         if v == media_type || v == '*/*'
29:           true
30:         else
31:           t, s, p = parse_media_type(v)
32:           t == type && (s == '*' || s == subtype) && (p == '' || params_match?(params, p))
33:         end
34:       }.sort_by {|v|
35:         # Most specific gets precedence.
36:         v.length
37:       }.reverse
